Model Definition with Parameters

model UserMetrics(region: String, startDate: String) = {
  from 'users.parquet'
  where region == region and created_date >= startDate
  select 
    user_id,
    name,
    age,
    email,
    created_date
  order by created_date desc
}

Complex Query with Aggregations

from UserMetrics('US', '2024-01-01')
join Orders o on o.user_id = _.user_id
group by age_group = case
  when age < 25 then 'Young'
  when age < 50 then 'Middle'
  else 'Senior'
end
agg
  total_users = count(distinct user_id),
  total_orders = count(o.order_id),
  avg_order_value = avg(o.amount),
  total_revenue = sum(o.amount)
having total_users > 100
order by total_revenue desc

Function Definition

def calculateRisk(balance: Double, creditScore: Int): String = {
  case
    when balance > 10000 and creditScore > 750 then 'Low'
    when balance > 5000 and creditScore > 650 then 'Medium'
    else 'High'
  end
}
